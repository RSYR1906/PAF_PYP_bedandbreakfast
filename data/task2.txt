Task 2.3:
db.listings.aggregate([
    {
        $unwind: "$reviews"
    },
    {
        $project: {
            _id: "$reviews._id",
            date: "$reviews.date",
            listing_id: "$reviews.listing_id",
            reviewer_name: {
                $replaceAll: {
                    input: "$reviews.reviewer_name",
                    find: ",",
                    replacement: ""
                }
            },
            comments: {
                $replaceAll: {
                    input: {
                        $replaceAll: {
                            input: "$reviews.comments",
                            find: "\n",
                            replacement: ""
                        }
                    },
                    find: "\r",
                    replacement: ""
                }
            }
        }
    },
    {
        $out: "reviews"
    }
]);
